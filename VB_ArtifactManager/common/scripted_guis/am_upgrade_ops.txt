AM_upgrade_artifact_set_id = {
	scope = character
	saved_scopes = { artifact }
	is_shown = { always = yes }
	effect = {
		remove_variable = AM_upgrade_op_selected_artifact
		remove_variable = AM_upgrade_op_gold_cost

		set_variable = {
			name = AM_upgrade_op_selected_artifact
			months = 3
			value = scope:artifact
		}
        if = {
            limit = {
                scope:artifact = { rarity = illustrious }
            }
            set_variable = {
                name = AM_upgrade_op_gold_cost
                months = 3
                value = 1000
            }
        }
        else_if = {
            limit = {
                scope:artifact = { rarity = famed }
            }
            set_variable = {
                name = AM_upgrade_op_gold_cost
                months = 3
                value = 2200
            }
        }
        else_if = {
            limit = {
                scope:artifact = { rarity = masterwork }
            }
            set_variable = {
                name = AM_upgrade_op_gold_cost
                months = 3
                value = 3300
            }
        }
        else = {
            set_variable = {
                name = AM_upgrade_op_gold_cost
                months = 3
                value = 5000
            }
        }
	}
}

AM_upgrade_artifact_closed = {
