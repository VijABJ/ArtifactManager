am_option_toggle_receiver_close_family = {
    scope = character
    is_shown = {
        NOT = {
            has_global_variable = am_option_receiver_close_family
        }
    }
    effect = {
        if = {
            limit = {
                NOT = {
                    has_global_variable = am_option_receiver_close_family
                }
            }
            set_global_variable = am_option_receiver_close_family
        }
        else = {
            remove_global_variable  =  am_option_receiver_close_family
        }

    }
}
